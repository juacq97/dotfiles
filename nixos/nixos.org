Importing the hardware-configuration file
#+begin_src nix
{ config, pkgs, ... }:

{
  imports =
    [ # Include the results of the hardware scan.
      ./hardware-configuration.nix
    ];
#+end_src

* Grub configuration
Use the GRUB 2 boot loader. No efi system
#+begin_src nix
  boot.loader.grub = {
    enable = true;
    version = 2;
    device = "/dev/sda"; 
  };
#+end_src
